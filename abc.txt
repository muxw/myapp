var fn = function(n) {
    var arr = [];
    if(n>31){
        n=31;
    }
    for(var i=0;i<n;i++){
        var rand = getRand(2,32);
        if(arr.indexOf(rand)!= -1){
            i--;
        }else{
            arr.push(rand)
        }
    }
    return arr;
}
function getRand(min, max) {
  min = Math.ceil(min);
  max = Math.floor(max);
  return Math.ceil(Math.random() * (max - min + 1)) + min;
}
